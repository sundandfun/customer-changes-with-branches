Customers can overwrite specific functions in customers branch

function main()
{
    return true;
}

// Return sum
function overwrite1 (a,b)
{
    // increase a by 1
    a = a+1;
    return a + b;
}

// Return dif
function overwrite2 (a,b)
{
    return a - b;
}

// return parameter
function noOverwrite (a)
{
    return a;
}
